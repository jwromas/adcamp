from flask import Flask, request, jsonify from flask_cors import CORS from openai import OpenAI app = Flask(__name__) CORS(app) client = OpenAI( base_url="http://127.0.0.1:1234/v1", api_key="lm-studio" ) MODEL = "mistral" @app.route("/route", methods=["POST"]) def route(): data = request.get_json(silent=True) message = data.get("message", "") prompt = f""" You are a routing AI. Decide which agent should handle this request. Agents: 1. analytics → website stats, users, sessions, tracking, data 2. market → SEO, ads, marketing, growth, content, branding Return ONLY one word: analytics OR market User message: {message} """ try: res = client.chat.completions.create( model=MODEL, messages=[ {"role": "system", "content": "You are a strict classifier."}, {"role": "user", "content": prompt} ] ) decision = res.choices[0].message.content.strip().lower() if "market" in decision: decision = "market" else: decision = "analytics" return jsonify({"agent": decision}) except Exception as e: return jsonify({"agent": "analytics", "error": str(e)}) if __name__ == "__main__": app.run(port=5052, debug=False)
