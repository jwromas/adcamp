from flask import Flask, request, jsonify from flask_cors import CORS from openai import OpenAI app = Flask(__name__) CORS(app) client = OpenAI( base_url="http://127.0.0.1:1234/v1", api_key="lm-studio" ) MODEL = "mistral" @app.route("/route", methods=["POST"]) def route(): data = request.get_json(silent=True) if not isinstance(data, dict): return jsonify({"agent": "analytics"}) message = data.get("message", "").lower().strip() prompt = f""" You are a STRICT routing system. You must choose ONLY ONE: - analytics = website data, users, traffic, sessions, dashboards, tracking - market = marketing, SEO, ads, promotion, growth, branding, content IMPORTANT RULES: - If message contains: promote, grow, advertise, SEO, marketing → ALWAYS choose market - If message is about increasing users or visibility → market - If message is about numbers or stats → analytics Return ONLY ONE WORD: analytics OR market Message: {message} """ try: res = client.chat.completions.create( model=MODEL, messages=[ {"role": "system", "content": "You are a strict classification router."}, {"role": "user", "content": prompt} ] ) decision = res.choices[0].message.content.lower() if "market" in decision: return jsonify({"agent": "market"}) else: return jsonify({"agent": "analytics"}) except Exception: return jsonify({"agent": "analytics"}) if __name__ == "__main__": app.run(port=5052, debug=False)
